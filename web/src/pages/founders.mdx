import Layout from "../components/Layout"
import PersonCard from "../components/PersonCard"
import Gallery from "../components/Gallery"
import { graphql } from "gatsby"
import { jsx } from "theme-ui"

<Layout>

# Founders

<br />

<Gallery n={4}>
{props.data.allAirtable.edges.map((edge, i) => {
  const person = edge.node.data
  return (
    person.Picture &&
    <PersonCard
      photoUrl={edge.node.data.Picture ? edge.node.data.Picture[0].thumbnails.large.url : ''}
      name={edge.node.data.Name}
      bio={edge.node.data.Bio}
      title={edge.node.data.Title}
    />
  )
})}
</Gallery>
</Layout>

export const pageQuery = graphql`
  query FoundersPageQuery {
    allAirtable(
      filter: { table: { eq: "Founders" } }
      sort: { order: ASC, fields: data___Website_Order }
    ) {
      edges {
        node {
          data {
            Name
            Bio
            Title
            Picture {
              thumbnails {
                large {
                  url
                }
              }
            }
          }
        }
      }
    }
  }
`
